FOREACH(SD ${SUBDIRS_BUILD_SEQ})
    FOREACH(G_SD ${GROUP_SUBDIRS})
        IF(G_SD STREQUAL ${SD})
            ADD_SUBDIRECTORY(${G_SD})
        ENDIF()
    ENDFOREACH()

    FOREACH(M_SD ${MODULE_SUBDIRS})
        IF(M_SD STREQUAL ${SD})
            ADD_SUBDIRECTORY(${M_SD})
        ENDIF()
    ENDFOREACH()
ENDFOREACH()
