# Project name
IF($ENV{BUILD_TYPE} MATCHES "DEBUG")
    PROJECT(${PROJECT_BASE_NAME}_d)
ELSE()
    PROJECT(${PROJECT_BASE_NAME})
ENDIF()

ADD_DEFINITIONS(${CUSTOM_DEFS})

INCLUDE_DIRECTORIES(BEFORE ${CUSTOM_INC_PATH})

# Project target
IF(PROJECT_BASE_TYPE STREQUAL "LIBRARY")
    ADD_LIBRARY(${PROJECT_NAME} SHARED ${SRC_LIST})

    SET(LIBRARY_OUTPUT_PATH $ENV{OUTPUT_BIN_BUILD})
ENDIF()

IF(PROJECT_BASE_TYPE STREQUAL "EXECUTABLE")
    ADD_EXECUTABLE(${PROJECT_NAME} ${SRC_LIST})

    SET(EXECUTABLE_OUTPUT_PATH $ENV{OUTPUT_BIN_BUILD})
ENDIF()

TARGET_LINK_LIBRARIES(${PROJECT_NAME} ${DEP_LIST})
